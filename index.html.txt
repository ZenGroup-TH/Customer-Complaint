    // NEW: ระบบเรียงหัวตาราง (Sorting)
    function sortTable(column) {
      if (currentSortCol === column) {
        sortAsc = !sortAsc; // สลับทิศทาง
      } else {
        currentSortCol = column;
        sortAsc = true;
      }
      
      // อัปเดตลูกศรบนหัวตาราง UI
      ['no', 'date', 'brand', 'branch'].forEach(col => {
        const th = document.getElementById(`th-${col}`);
        if(th) {
          th.classList.remove('sort-asc', 'sort-desc');
          if (col === currentSortCol) {
            th.classList.add(sortAsc ? 'sort-asc' : 'sort-desc');
          }
        }
      });

      renderTable();
    }

    function parseDateForSort(dateStr) {
      if (!dateStr || dateStr === '-') return 0;
      const parts = dateStr.split('/');
      if (parts.length === 3) {
        // เปลี่ยน 31/12/2026 เป็น 20261231 เพื่อเรียงตัวเลข
        return parseInt(parts[2] + parts[1].padStart(2, '0') + parts[0].padStart(2, '0'));
      }
      return 0;
    }

function initializeDashboard(jsonStr) {
      allData = JSON.parse(jsonStr);
      if (allData.error) { alert(allData.error); return; }

      allData.forEach(item => {
        // อัปเดตสถานะการปิดเคส
        item.statusFilter = (item.closedBy && item.closedBy.trim() !== '') ? item.closedBy : 'กำลังดำเนินเรื่อง';
        // (ลบลอจิก .split('/') อันเก่าออก เพราะรับค่า monthNum สำเร็จรูปจากหลังบ้านแล้ว)
      });

      filteredData = [...allData];

      populateCheckbox('branchOptions', 'branchBtnText', 'ทุกสาขา', 'branch');
      populateCheckbox('channelOptions', 'channelBtnText', 'ทุกช่องทาง', 'channel');

      document.getElementById('loading').classList.add('hidden');

    function populateCheckbox(containerId, btnId, defaultText, key) {
      const container = document.getElementById(containerId);
      container.innerHTML = '';
      
      let allValues = [];
      allData.forEach(item => {
        let val = item[key];
        if (Array.isArray(val)) allValues.push(...val);
        else allValues.push(val);
      });

      const uniqueValues = [...new Set(allValues)].filter(val => val && val !== '-').sort();
      
      if (uniqueValues.length > 0) {
        const selectAllLabel = document.createElement('label');
        selectAllLabel.className = 'flex items-center space-x-2 px-2 py-2 hover:bg-slate-100 cursor-pointer text-sm font-semibold text-slate-800 rounded transition-colors border-b border-slate-200 mb-1';
        selectAllLabel.innerHTML = `<input type="checkbox" class="rounded border-slate-300 text-indigo-600 focus:ring-indigo-500 w-4 h-4 select-all-cb" onchange="toggleSelectAll('${containerId}', this, '${btnId}', '${defaultText}')"><span class="truncate">เลือกทั้งหมด</span>`;
        container.appendChild(selectAllLabel);
      }

      uniqueValues.forEach(val => {
        const label = document.createElement('label');
        label.className = 'flex items-center space-x-2 px-2 py-1.5 hover:bg-indigo-50 cursor-pointer text-sm text-slate-700 rounded transition-colors';
        label.innerHTML = `<input type="checkbox" value="${val}" class="rounded border-slate-300 text-indigo-600 focus:ring-indigo-500 w-4 h-4 item-cb" onchange="handleFilterChange('${containerId}', '${btnId}', '${defaultText}')"><span class="truncate">${val}</span>`;
        container.appendChild(label);
      });
    }

    function populateMonthCheckbox() {

    function toggleSelectAll(containerId, selectAllElement, btnId, defaultText) {
      const isChecked = selectAllElement.checked;
      const checkboxes = document.querySelectorAll(`#${containerId} .item-cb`);
      checkboxes.forEach(cb => { cb.checked = isChecked; });
      handleFilterChange(containerId, btnId, defaultText);
    }

    function getSelectedValues(containerId) {
      const checkedBoxes = document.querySelectorAll(`#${containerId} .item-cb:checked`);
      return Array.from(checkedBoxes).map(cb => cb.value);
    }

    function handleFilterChange(containerId, btnId, defaultText) {
      const allBoxes = document.querySelectorAll(`#${containerId} .item-cb`);
      const checkedBoxes = document.querySelectorAll(`#${containerId} .item-cb:checked`);
      const selectAllBox = document.querySelector(`#${containerId} .select-all-cb`);
      
      if (selectAllBox) selectAllBox.checked = (allBoxes.length > 0 && allBoxes.length === checkedBoxes.length);

      const selected = Array.from(checkedBoxes).map(cb => cb.value);
      const btn = document.getElementById(btnId);
      
      if (selected.length === 0 || selected.length === allBoxes.length) btn.textContent = `ทั้งหมด ${defaultText}`;
      else if (selected.length === 1) btn.textContent = checkedBoxes[0].nextElementSibling.textContent;
      else btn.textContent = `เลือกแล้ว ${selected.length} รายการ`;
      
      applyFilters();
    }

    function applyFilters() {
      const brands = getSelectedValues('brandOptions');
      const selectedCats = getSelectedValues('categoryOptions');
      const selectedTopics = getSelectedValues('topicOptions');     
      const branches = getSelectedValues('branchOptions');  
      const channels = getSelectedValues('channelOptions');
      const resps = getSelectedValues('responsibleOptions');
      const months = getSelectedValues('monthOptions');
      const statuses = getSelectedValues('statusOptions');

      filteredData = allData.filter(item => {
        const matchBrand = brands.length === 0 || brands.includes(item.brand);
        const matchBranch = branches.length === 0 || branches.includes(item.branch); 
        const matchChannel = channels.length === 0 || channels.includes(item.channel);
        const matchResp = resps.length === 0 || resps.includes(item.responsible);
        const matchMonth = months.length === 0 || months.includes(item.monthNum.toString());
        const matchStatus = statuses.length === 0 || statuses.includes(item.statusFilter);
        const matchCat = selectedCats.length === 0 || item.categories.some(c => selectedCats.includes(c));
        const matchTopic = selectedTopics.length === 0 || item.topics.some(t => selectedTopics.includes(t));       

        return matchBrand && matchCat && matchTopic && matchBranch && matchChannel && matchResp && matchMonth && matchStatus;
      });
      
      renderDashboard();
    }

function calculateKPIs() {
      const validCases = filteredData.filter(item => item.no && item.no.toString().trim() !== '');
      const total = validCases.length;
      const closedCasesList = validCases.filter(item => item.closedBy && item.closedBy !== '');
      const closed = closedCasesList.length;
      const pending = total - closed;
      const rate = total > 0 ? ((closed / total) * 100).toFixed(1) : 0;
      
      document.getElementById('kpiTotal').textContent = total.toLocaleString();
      document.getElementById('kpiClosed').textContent = closed.toLocaleString();
      document.getElementById('kpiPending').textContent = pending.toLocaleString();
      document.getElementById('kpiRate').textContent = `(${rate}%)`;

      // 🌟 NEW: คำนวณ SLA แปลงนาทีเฉลี่ยเป็น "วัน-ชั่วโมง-นาที"
      const slaCases = closedCasesList.filter(item => item.slaMinutes !== null && item.slaMinutes !== undefined);
      if (slaCases.length > 0) {
        const totalMinutes = slaCases.reduce((sum, item) => sum + item.slaMinutes, 0);
        const avgMinutes = Math.round(totalMinutes / slaCases.length); // ได้ค่าเฉลี่ยเป็นนาที
        
        // แยกตัวประกอบคณิตศาสตร์
        const d = Math.floor(avgMinutes / (24 * 60));
        const h = Math.floor((avgMinutes % (24 * 60)) / 60);
        const m = avgMinutes % 60;
        
        let slaText = "";
        if (d > 0) slaText += `${d} วัน `;
        if (h > 0) slaText += `${h} ชม. `;
        slaText += `${m} นาที`;
        
        document.getElementById('kpiSLA').textContent = slaText;
      } else {
        document.getElementById('kpiSLA').textContent = 'N/A';
      }

      const brandCounts = {};
      let topBrandName = '-';
      let maxCount = 0;
      validCases.forEach(item => {
        if(item.brand && item.brand !== 'ไม่ระบุแบรนด์') {
          brandCounts[item.brand] = (brandCounts[item.brand] || 0) + 1;
          if(brandCounts[item.brand] > maxCount) {
            maxCount = brandCounts[item.brand];
            topBrandName = item.brand;
          }
        }
      });
      document.getElementById('kpiTopBrand').textContent = topBrandName + (maxCount > 0 ? ` (${maxCount} เคส)` : '');
    }

    function getTop10Data(dataArray, key) {
      const counts = {};
      dataArray.forEach(item => {
        let values = item[key];
        if (!values) values = ['ไม่ระบุ'];
        if (!Array.isArray(values)) values = [values];
        values.forEach(val => { counts[val || 'ไม่ระบุ'] = (counts[val || 'ไม่ระบุ'] || 0) + 1; });
      });
      return Object.entries(counts).sort((a, b) => b[1] - a[1]).slice(0, 10);
    }

    function renderTable() {
      const tbody = document.getElementById('tableBody');
      tbody.innerHTML = '';
      
      if (filteredData.length === 0) {
        tbody.innerHTML = `<tr><td colspan="7" class="px-4 py-8 text-center text-slate-400 font-light">ไม่พบข้อมูลที่ตรงกับตัวกรองสิทธิ์นี้</td></tr>`;
        return;
      }
      
      // การเรียงลำดับตาราง
      let tableRows = [...filteredData];
      tableRows.sort((a, b) => {
        let valA = a[currentSortCol] || '';
        let valB = b[currentSortCol] || '';
        
        if (currentSortCol === 'no') {
          valA = parseInt(valA) || 0;
          valB = parseInt(valB) || 0;
        } else if (currentSortCol === 'date') {
          valA = parseDateForSort(valA);
          valB = parseDateForSort(valB);
        } else {
          valA = valA.toString().toLowerCase();
          valB = valB.toString().toLowerCase();
        }

        if (valA < valB) return sortAsc ? -1 : 1;
        if (valA > valB) return sortAsc ? 1 : -1;
        return 0;
      });
      
      tableRows.forEach(item => {
        const isClosed = item.closedBy && item.closedBy !== '';
        const badgeClass = isClosed ? 'bg-emerald-50 text-emerald-700 border border-emerald-200' : 'bg-amber-50 text-amber-700 border border-amber-200';
        const badgeText = isClosed ? `ปิดเคส (${item.closedBy})` : 'กำลังดำเนินเรื่อง';
        const catText = item.categories.join(', ');
        
        const tr = document.createElement('tr');
        tr.className = 'hover:bg-slate-50 transition-colors border-b border-slate-100';
        tr.innerHTML = `
          <td class="px-4 py-3 text-center text-xs font-mono font-bold text-slate-400">${item.no}</td>
          <td class="px-4 py-3 text-xs text-slate-500 whitespace-nowrap">${item.date}</td>
          <td class="px-4 py-3 font-semibold text-slate-700 text-xs">${item.brand}</td>
          <td class="px-4 py-3 text-xs text-slate-600 truncate max-w-[140px]" title="${item.branch}">${item.branch}</td>
          <td class="px-4 py-3 text-xs text-slate-500 truncate max-w-[160px]" title="${catText}">${catText}</td>
          <td class="px-4 py-3 text-xs text-slate-600 max-w-sm truncate" title="${item.complaint}">${item.complaint}</td>
          <td class="px-4 py-3 text-center">
            <span class="inline-flex items-center px-2 py-0.5 rounded-full text-xs font-medium ${badgeClass}">${badgeText}</span>
          </td>
        `;
        tbody.appendChild(tr);
      });
    }
